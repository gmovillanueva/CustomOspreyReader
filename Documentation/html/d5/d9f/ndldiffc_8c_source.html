65  unsigned char *input_ptr = input; /* points to current byte in "input" */
66  LONG *output_ptr = output; /* points to current channel in "output" */
67  unsigned char *end_input_ptr = input+bytes; /* points past last byte in "input" */
68  LONG *end_output_ptr = output+(*channels); /* points past last channel in "output" */
69  SHORT short_value=0;
70 /*
71 * Set up a loop over the number of bytes in the input/channels in output:
72 * both of these should go to 0 at the end!
73 *
74 * This loop is a bit wierdly constructed: the idea is that the most common
75 * case should be the 8 bit delta value, so the code should (assumming the
76 * compiler doesn't do anything too smart) fall thru the tests most of the
77 * time; the branches should occur less frequently.
78 */
79 
80  while(input_ptr < end_input_ptr && output_ptr < end_output_ptr) {
81 
82  /* Look for 3 byte format: 7F followed by 16 bit delta */
83 
84  if(*input_ptr == (unsigned char ) 0x7F) goto do_16bit;
85 
86  /* Look for 5 byte format: 80 followed by 32 bit absolute value */
87 
88  else if(*input_ptr == (unsigned char) 0x80) goto do_32bit;
89 #if 0
90  else if(*input_ptr == (unsigned char) 0x81) goto do_RLE;
91 #endif
92  /* What it must be is an 8 bit delta */
93 
94  current_channel_value += *((char *) input_ptr);
95  input_ptr++;
96 
97  /* Store the current channel value and advance to the next channel */
98 
99 store: *output_ptr = current_channel_value;
100  output_ptr++;
101 
102  }
103 
104 /*
105 * Make sure that we used up all the input bytes and filled up the output
106 * array and return.
107 */
108 
109  *channels = output_ptr - output;
110  if(input_ptr != end_input_ptr || output_ptr != end_output_ptr) return GEN_COMPRESSION_FAILURE;
111  return 0;

121  current_channel_value += short_value;

124
