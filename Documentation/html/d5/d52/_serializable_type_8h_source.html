14  //This template may be used to serializing most data types
15  //such as any of the following:
16  // -) Native types: int, uint, short, ushort, double, float, long, ulong, char, uchar
17  // -) Structure types: Structures must not contain references (pointers)
18  //
19  //The serialized data will contain a metadata section that defines the information
20  //necessary to deserialize the type. See MetaData.h for all the information in the
21  //metadata section.
22  //
23  //The template may be used to serialize other types provided you subclass this template.
24  //IMPORTANT: If you serialize custom types you must defined its type code uniquely see
25  // SerializedObject.h for an enumeration you can add to
26  //
27  //See some example serializable types at the bottom of this file
28  //
29  template <class Value_t>
30  class SERIALIZATION_IMPORT_USAGE SerializableType {
31  protected:
32  Value_t _Value; //The value to serialize
33 
34  ULONG GetValueSize() {return sizeof(Value_t);}
35  ULONG SerializeValue(Canberra::Utility::Core::ByteStream &Str, const MetaData &MD, const LONG Offset=0) {
36  LONG Pos=Offset;
37  LONG NumCpy=0;
38  size_t DataSize=const_cast<MetaData &>(MD).GetDataSize();
39  Canberra::Utility::Core::ByteStream Buf(DataSize);
40 
41  memcpy(const_cast<char *>(Buf.GetBuffer()), &_Value, DataSize);
42  SWAPNBYTES(Buf.GetBuffer(), DataSize);
43 
44  memcpy(const_cast<char *>(Str.GetBuffer()+Pos), Buf.GetBuffer(), NumCpy=DataSize); Pos+=NumCpy;
45  return Pos-Offset;
46  }
47  ULONG DeserializeValue(Canberra::Utility::Core::ByteStream &Str, const MetaData &MD, const LONG Offset=0) {
48  LONG Pos=Offset;
49  LONG NumCpy=0;
50  size_t DataSize=const_cast<MetaData &>(MD).GetDataSize();
51 
52  memcpy(&_Value, const_cast<char *>(Str.GetBuffer()+Pos), NumCpy=DataSize); Pos+=NumCpy;
53  SWAPNBYTES(&_Value, DataSize);
54  return Pos-Offset;
55  }
56 
57  public:
58  SerializableType(Value_t v) {PutValue(v); }
59  SerializableType() {memset(&_Value, 0, sizeof(_Value));}
60  virtual ~SerializableType() {}
61 
62  //Accessor to the value to be serialized
63  Value_t GetValue() {return _Value;}
64  void PutValue(Value_t v) {memcpy(&_Value, &v, sizeof(v));}
