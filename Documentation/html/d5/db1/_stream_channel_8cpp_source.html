135 ULONG Canberra::Protocols::Lynx::StreamChannel::Receive(ByteStream &Stream, ULONG *Recv, LONG Flags)
136 {
137  LockManager LM(&_Lock);
138 
139  if (Recv) *Recv=0;
140  if (!_Channel.GetIsOpen()) {
141  ULONG Stat=(ULONG) E_FAIL;
142  DEBUG_ERROR_MESSAGE(_T("StreamChannel"), _T("Receive"), _T("Socket is not open."), Stat);
143  return Stat;
144  }
145  if (NULL == Stream.GetBuffer()) {
146  ULONG Stat=(ULONG) E_INVALIDARG;
147  DEBUG_ERROR_MESSAGE(_T("StreamChannel"), _T("Receive"), _T("Buffer is null."), Stat);
148  return Stat;
149  }
150 
151  ULONG NBytesRecv=0, NBufBytes=0;
152  ULONG BufSize = Stream.GetSize();
153  const char * const Buf = Stream.GetBuffer();
154  while(NBufBytes < BufSize) {
155  if (_Abort) {
156  //
157  //We are closing the socket and opening a new one
158  //because an abort will cause the data stream to
159  //be out of sync. Once out of sync it will become
160  //very complicated to resync device with client
161  //Therefore, simply close the connection and open
162  //a new one.
163  //
164 
165  //Close the socket
166  LM.Unlock();
167  Close();
168 
169  //Open a new one for fresh communications
170  Open(_ClientEP.GetAddressString(), _DevEP.GetAddressString());
171  return 0;
172  }
173 
174  LONG Stat=recv(_Channel.GetHandle(), const_cast<char *>(Buf+NBufBytes), BufSize-NBufBytes, Flags);
175  if (SOCKET_ERROR == Stat) {
176  ULONG Stat = WSAGetLastError();
177  if (PLATFORM_INDEP_WOULDBLOCK == Stat) {
178 #ifdef WIN32
179  Sleep(0); //Yield
180 #else
181  sched_yield();
182 #endif
183  continue;
184  }
185  DEBUG_ERROR_MESSAGE(_T("StreamChannel"), _T("Receive"), _T("Error recv."), Stat);
186  return Stat;
187  }
188  else if (Stat == 0) break;
189 
190  NBytesRecv += Stat;
191  NBufBytes += Stat;
192  }
193  if (Recv) *Recv=NBytesRecv;
194  return 0;
195 }

223  ULONG Stat = Receive(Str, &Size);

237  LONG PSize=MsgSize-sizeof(MsgSize);
238  if (PSize < 0) {

241  return Stat;

245  Stat = Receive(Frag, &Size);
246  if (_Abort) {
247  DEBUG_TYPED_MESSAGE(L"StreamChannel", L"ReceiveCommand", L"Abort flag set operation aborted", Canberra::Utility::Core::Diagnostics::Warning);
248  return (ULONG) E_ABORT;
249  }

252  DEBUG_ERROR_MESSAGE(L"StreamChannel", L"ReceiveCommand", L"Error while receiving device response body.", Stat);
253  return Stat;
254  }
255 
256  Stat=0;
257  SerializableObject *Obj=NULL;
258  Str.Copy(Frag, 0, sizeof(MsgSize));
259  try {
260  Obj = _MessageFactory.DeserializeFromMessage(Str);
261  }
262  CATCHTRACE(L"StreamChannel", L"ReceiveCommand", Stat);
263  if (Stat) {
264  delete Obj;
265  return Stat;
266  }
267 
268  *Resp = dynamic_cast<CommandData *>(Obj);
269  if (!Resp) {
270  delete Obj;
271  Stat = GEN_INVALID_RESPONSE;
272  DEBUG_ERROR_MESSAGE(L"StreamChannel", L"ReceiveCommand", L"Error device response is invalid. Argument is not a ListData nor TlistData type.", Stat);
273  goto EXIT;
274  }
275 
276 EXIT:
