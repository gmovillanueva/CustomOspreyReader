140  static T Convert(VariantType _Type, union NativeValues _uVal, const String &_strVal)
141  {
142  T result = NULL;
143 
144  switch(_Type) {
145  case Int64:
146  return (T) _uVal.i64Val;
147  case Uint64:
148  return (T) _uVal.ui64Val;
149  case Bool:
150  return (T) (_uVal.boolVal ? 1 : 0);
151  case Ubyte:
152  return (T) _uVal.ubVal;
153  case Ushort:
154  return (T) _uVal.usVal;
155  case Ulong:
156  return (T) _uVal.ulVal;
157  case Byte:
158  return (T) _uVal.bVal;
159  case Short:
160  return (T) _uVal.sVal;
161  case Long:
162  return (T) _uVal.lVal;
163  case Float:
164  return (T) _uVal.fVal;
165  case Double:
166  return (T) _uVal.dVal;
167  case DateTime:
168  return (T) _uVal.int64DateVal;
169  case Null:
170  return (T) 0;
171  case Wstring:
172  {
173  //
174  // TODO : There is a defect here ... we are not handling signed vs unsigned ints
175  //
176  if ((typeid(T) == typeid(float)) || (typeid(T) == typeid(double)))
177  {
178  wchar_t *StopChar=NULL;
179  if (_strVal.GetLength()) return (T) wcstod(const_cast<String &>(_strVal), &StopChar);
180  else return (T) 0;
181  }
182 
183  else if (typeid(T) == typeid(bool))
184  {
185  //
186  // We will support various text representations
187  // of boolean values. Text starting with the
188  // characters 'y' or 't' will cause the result
189  // to be TRUE, and text starting with the
190  // characters 'n' or 'f' will cause the result
191  // to be false.
192  //
193  // If neither case is true, then we will parse
194  // the value as a number, and return TRUE for
195  // non-zero results. Note that this will cause
196  // any other non-numeric text to return FALSE.
197  //
198  if (_strVal.GetLength() == 0)
199  {
200  return false;
201  }
202 
203  wchar_t c = ((const wchar_t *)_strVal)[0];
204  if (c == L'y' || c == L'Y' || c == L'T' || c == L't')
205  {
206  return true;
207  }
208 
209  else if (c == L'n' || c == L'N' || c == L'F' || c == L'f')
210  {
211  return false;
212  }
213 
214  return atol(const_cast<String &>(_strVal)) != 0 ? true : false;
215  }
216 
217  else
218  {
219  //
220  // parse the result into an __i64, then cast
221  // it down to T
222  //
223  LONGLONG ival = 0;
224  if (_strVal.GetLength()) {
225  //wstring theVal((const wchar_t*)_strVal);
226  //wistringstream is(theVal);
227  //is >> ival;
228  from_string<LONGLONG>(ival,std::string(_strVal),std::dec);
229  return (T) ival;
230  }
231  else return (T) 0;
232  }
233  }
234 
235  //
236  // Now handle the Array types
237  //
238  case aUbyte:
239  case aUshort:
240  case aUlong:
241  case aByte:
242  case aShort:
243  case aLong:
244  case aFloat:
245  case aDouble:
246  case aWstring:
247  case aInt64:
248  case aUint64:
249  case aBool:
250  case aDateTime:
251  DEBUG_ERROR_MESSAGE(L"Variant", L"ConvertTo", L"Conversion from array types is not currently supported", E_FAIL);
252  break;
253 
254 
255  default:
256  DEBUG_ERROR_MESSAGE(L"Variant", L"ConvertTo", String(L"Cannot convert unknown type. Type code is: ") + String(_Type), E_FAIL);
257 
258  break;
259  }
260 
261  return result;
