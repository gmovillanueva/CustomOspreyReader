624

627 
628  _MessageFactory.SerializeToMessage(&Cmd, Str);

630  if (Stat || !Size) {
631  if (!Stat) Stat = WSAECONNRESET;

634  }

637  Str.Clear();
638  Str.Allocate(sizeof(MsgSize));
639 
640  //Get the size of the response message (1'st 4 bytes)

645  return Stat;
646  }
647

649  memcpy(&MsgSize, Str.GetBuffer(), sizeof(MsgSize));
650  SWAPNBYTES(&MsgSize, 4);
651  Canberra::Utility::Core::ByteStream Frag;
652  LONG PSize=MsgSize-sizeof(MsgSize);
653  if (PSize < 0) {
654  Stat = (ULONG) E_FAIL;

656  return Stat;
657  }

664  return Stat;
665  }
666

668  SerializableObject *Obj=NULL;
669  Str.Copy(Frag, 0, sizeof(MsgSize));
670  try {
671  Obj = _MessageFactory.DeserializeFromMessage(Str);
672  *Resp = dynamic_cast<CommandData *>(Obj);
673  }
674  CATCHTRACE(L"Device", L"Control", Stat);
675 
676  if (Stat) {
677  delete Obj;
678  Obj=NULL;
679  return Stat;
680  }

683  Obj = NULL;
