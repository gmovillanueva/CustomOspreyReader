878  VARIANT args;

881  //Make sure we own the input
882  lynx->LockInput(user, pwd, input);
883  }
884  catch(...) {}
885 
886  try {
887  //Stop acquisition
888  lynx->Control(DevCntl::Stop, input, &args);
889  }
890  catch(...) {}
891 
892  try {
893  //Abort acquisition (only needed for MSS or MCS collections)
894  lynx->Control(DevCntl::Abort, input, &args);
895  }
896  catch(...) {}
897 
898  _variant_t disable((int) 0);
899  try {
900  //Stop SCA collection
901  lynx->PutParameter(DevCntl::Input_SCAstatus, input, &disable);
902  }
903  catch(...) {}
904 
905  try {
906  //Stop aux counters collection
907  lynx->PutParameter(DevCntl::Counter_Status, input, &disable);
908  }
909  catch(...) {}
910  try {
911  // see if the osprey was powered off
912  variant_t OspreyTime = lynx->GetParameter(DevCntl::Input_StartTime,input);
913  // this is the initial start time is less than january 2,2009
914  variant_t InitialOspreyTime = "2009-01-02 00:00:00";
915  InitialOspreyTime.ChangeType(VT_DATE);
916  if (static_cast<double>(OspreyTime) <= static_cast<double>(InitialOspreyTime))
917  {
918  // get the current time from the local computer in the utc format
919  SYSTEMTIME systemTime;
920  variant_t currentTime;
921  GetSystemTime(&systemTime);
922  // convert to variant time as this is what the osprey expects.
923  SystemTimeToFileTime(&systemTime, reinterpret_cast<LPFILETIME>(&currentTime.llVal));
924  currentTime.vt = VT_I8;
925  lynx->LockInput(user, pwd, static_cast<short>(0));
926  try
927  {
928  // set the start time of the next acquisitio
929  lynx->PutParameter(DevCntl::System_DateTime,0,&currentTime);
930  }
931  catch(...)
932  {
